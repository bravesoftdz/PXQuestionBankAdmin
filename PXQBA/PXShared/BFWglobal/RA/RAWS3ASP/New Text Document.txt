SELECT DISTINCT 	tblPackage.PackageID, 	tblPackage.Description, 	tblPackage.Type, 	tblSiteAssignments.AssignmentID, 	tblSiteAssignments.SiteID, 	tblSiteKey.BaseURL, 	tblLevelType.LevelOfAccess FROM	tblPackage WITH (nolock) 	INNER JOIN tblSiteAssignments (nolock) 		ON tblPackage.PackageID = tblSiteAssignments.PackageID 	LEFT JOIN tblSiteKey WITH (nolock) 		ON tblSiteKey.SiteID = tblSiteAssignments.SiteID 	INNER JOIN tblLevelType WITH (nolock) 		ON tblLevelType.LevelID = tblSiteAssignments.LevelID WHERE tblSiteAssignments.SiteID IN (24479,24487,24049) OR tblPackage.PackageID IN (933) ORDER BY tblPackage.PackageID, tblSiteAssignments.SiteID, tblLevelType.LevelOfAccess DESC
